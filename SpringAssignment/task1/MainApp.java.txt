public class MainApp {
    public static void main(String[] args) {
        ApplicationContext context = new ClassPathXmlApplicationContext("applicationContext.xml");

        PolicyService policyService = context.getBean(PolicyService.class);

        // Use policy service to perform operations
        List<Policy> policies = policyService.getAllPolicies();
        System.out.println(policies);

        Policy policy = policyService.getPolicyById(1);
        System.out.println(policy);

        policyService.createPolicy(new Policy(1, "New Policy", "This is a new policy"));
        policyService.updatePolicy(new Policy(1, "Updated Policy", "This is an updated policy"));
        policyService.deletePolicy(1);
    }
}
